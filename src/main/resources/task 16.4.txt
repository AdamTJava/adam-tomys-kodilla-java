select ISS.SUMMARY, ISS.ISSUE_DESCRIPTION, ISL.TASKLIST_NAME
from ISSUES ISS
join ISSUESLIST ISL on ISS.ISSUESLIST_ID = ISL.ID;

select ISS.SUMMARY, ISS.ISSUE_DESCRIPTION, ISL.TASKLIST_NAME, U.FIRSTNAME as ASSIGNED_USER_NAME, U.LASTNAME as ASSIGNED_USER_LASTNAME
from ISSUES ISS
join ISSUESLIST ISL on ISS.ISSUESLIST_ID = ISL.ID
join users U on ISS.USER_ID_ASSIGNEDTO = U.ID;

select ISS.USER_ID_ASSIGNEDTO, count(*) as ISSUES_NUMBER, U.FIRSTNAME, U.LASTNAME
from ISSUES ISS
join users U on ISS.USER_ID_ASSIGNEDTO = U.ID
group by USER_ID_ASSIGNEDTO
having count(*) > 0;
